# Web framework + ASGI server.
fastapi==0.115.0
uvicorn[standard]==0.32.0

# JWT verification. The [crypto] extra pulls in `cryptography` for
# RS256 signature verification. Explicitly required -- without it
# pyjwt refuses RS256 (only HS256 by default).
pyjwt[crypto]==2.9.0

# AWS SDK.
boto3==1.35.36

# Postgres v3 driver. Supports RDS IAM auth via a password callable,
# which is how we mint short-lived auth tokens for the RDS Proxy.
psycopg[binary]==3.2.3

# Schema + config.
pydantic==2.9.2
pydantic-settings==2.5.2

# Test stack.
httpx==0.27.2
pytest==8.3.3
pytest-asyncio==0.24.0
moto==5.0.16
